🚦 Traffic Sign Classifier (Capstone)

Welcome to your final AI project. Your goal is to build a Convolutional Neural Network (CNN) using TensorFlow/Keras that can classify 43 different traffic signs from the German Traffic Sign Benchmark (GTSRB).

📂 Project Structure

  • dataset/ - Contains train/test images (Download full dataset separately).
  • model.py - Define your CNN architecture here.
  • main.py - Training loop, data loading, and evaluation script.
  • utils.py - Helper functions for image preprocessing.

🎯 Success Criteria

You must achieve at least 95% accuracy on the test set to pass this milestone.

🚀 Quick Start

pip install -r requirements.txt
python main.py
import os
import numpy as np
import tensorflow as tf
from sklearn.model_selection import train_test_split
from model import create_traffic_model
from utils import load_and_preprocess_data

# --- CONFIGURATION ---
EPOCHS = 15
BATCH_SIZE = 64
DATA_DIR = './dataset'

def main():
    # 1. Load Data
    print(f"Loading images from {DATA_DIR}...")
    images, labels = load_and_preprocess_data(DATA_DIR)
    
    # 2. Split Data
    X_train, X_test, y_train, y_test = train_test_split(
        images, labels, test_size=0.2, random_state=42
    )
    
    # 3. Initialize Model
    model = create_traffic_model(input_shape=X_train.shape[1:])
    model.summary()
    
    # 4. Train
    print("Starting training...")
    history = model.fit(
        X_train, y_train,
        epochs=EPOCHS,
        batch_size=BATCH_SIZE,
        validation_data=(X_test, y_test)
    )
    
    # 5. Evaluate & Save
    loss, acc = model.evaluate(X_test, y_test)
    print(f"Final Test Accuracy: {acc*100:.2f}%")
    
    if acc > 0.95:
        model.save('traffic_classifier.h5')
        print("Model saved successfully!")
    else:
        print("Accuracy too low. Tune hyperparameters and retrain.")

if __name__ == "__main__":
    main()

from tensorflow.keras.models import Sequential
from tensorflow.keras.layers import Conv2D, MaxPool2D, Dense, Flatten, Dropout, BatchNormalization

def create_traffic_model(input_shape):
    """
    Constructs a CNN architecture optimized for traffic sign recognition.
    Input Shape: (30, 30, 3)
    Output: 43 classes (Softmax)
    """
    model = Sequential([
        # Convolutional Block 1
        Conv2D(32, (5,5), activation='relu', input_shape=input_shape),
        Conv2D(32, (5,5), activation='relu'),
        MaxPool2D(pool_size=(2, 2)),
        Dropout(0.25),

        # Convolutional Block 2
        Conv2D(64, (3,3), activation='relu'),
        Conv2D(64, (3,3), activation='relu'),
        MaxPool2D(pool_size=(2, 2)),
        Dropout(0.25),

        # Fully Connected Layers
        Flatten(),
        Dense(256, activation='relu'),
        Dropout(0.5),
        
        # Output Layer
        Dense(43, activation='softmax')
    ])

    model.compile(
        loss='categorical_crossentropy', 
        optimizer='adam', 
        metrics=['accuracy']
    )
    
    return model

import cv2
import os
import numpy as np
from tensorflow.keras.utils import to_categorical

def load_and_preprocess_data(data_dir):
    """
    Loads images from 43 subdirectories, resizes them to 30x30,
    and normalizes pixel values.
    """
    images = []
    labels = []
    
    classes = 43
    
    for i in range(classes):
        path = os.path.join(data_dir, str(i))
        images_list = os.listdir(path)
        
        for img_name in images_list:
            try:
                img_path = os.path.join(path, img_name)
                image = cv2.imread(img_path)
                image = cv2.resize(image, (30,30))
                images.append(image)
                labels.append(i)
            except Exception as e:
                print(f"Error loading image: {e}")
                
    images = np.array(images)
    labels = np.array(labels)
    
    # Normalize pixel values (0-255 -> 0-1)
    images = images.astype('float32') / 255.0
    
    # One-hot encode labels
    labels = to_categorical(labels, classes)
    
    return images, labels
